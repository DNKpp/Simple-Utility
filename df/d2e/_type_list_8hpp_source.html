102 template <class T>
103 concept type_list = requires
104 {
105 typename std::tuple_size<T>::type;
106 { std::tuple_size_v<T> } -> std::convertible_to<std::size_t>;
107 }
108 && 0u <= std::tuple_size_v<T>
109 && type_list::detail::checkIndices<T>;
110
115 template <class T>
116 concept populated_type_list = type_list<T>
117 && 0u < std::tuple_size_v<T>;
