28 template <std::size_t index>
29 struct get_fn
30 {
31 template <class Tuple>
32 [[nodiscard]]
33 constexpr decltype(auto) operator ()(Tuple&& tuple) const noexcept
34 {
35 using std::get;
36 return get<index>(std::forward<decltype(tuple)>(tuple));
37 }
38 };
