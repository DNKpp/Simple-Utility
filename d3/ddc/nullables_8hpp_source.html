123 template <class T>
124 concept input_nullable = requires
125 {
126 typename nullable_value_t<T>;
127 nullable_null_v<T>;
128 }
129 && concepts::weakly_equality_comparable_with<T, decltype(nullable_null_v<T>)>;

139 template <class T>
140 concept nullable = input_nullable<T>
141 && concepts::initializes<std::remove_cvref_t<decltype(nullable_null_v<T>)>, std::remove_cvref_t<T>>
142 && std::is_assignable_v<std::remove_cvref_t<T>&, decltype(nullable_null_v<T>)>;

347 template <input_nullable UNullable, class UFunc>
348 [[nodiscard]]
349 constexpr TNullable operator()(UNullable&& closure, UFunc&& func)
350 {
351 if (closure != nullable_null_v<UNullable>)
352 {
353 return std::forward<UNullable>(closure);
354 }
355
356 if constexpr (std::same_as<void, std::invoke_result_t<UFunc>>)
357 {
358 std::invoke(std::forward<UFunc>(func));
359 return nullable_null_v<UNullable>;
360 }
361 else
362 {
363 return std::invoke(std::forward<UFunc>(func));
364 }
365 }

381 template <input_nullable TNullable, class TFunc>
382 struct and_then_func_t<TNullable, TFunc>
383 {
391 template <input_nullable UNullable, class UFunc>
392 [[nodiscard]]
393 constexpr auto operator()(UNullable&& closure, UFunc&& func)
394 {
395 static_assert
396 (
397 std::invocable<UFunc, detail::dereference_type_t<UNullable>>,
398 "TFunc must accept the value returned by TNullables as parameter."
399 );
400
401 using return_t = std::invoke_result_t<UFunc, detail::dereference_type_t<UNullable>>;
402 static_assert(nullable<return_t>, "TFunc must return a nullable type.");
403
404 if (closure != nullable_null_v<UNullable>)
405 {
406 return std::invoke(std::forward<UFunc>(func), value_unchecked(std::forward<UNullable>(closure)));
407 }
408 return return_t{ nullable_null_v<return_t> };
409 }
