121 template <class T>
122 concept input_nullable = requires
123 {
124 typename nullable_value_t<T>;
125 nullable_null_v<T>;
126 }
127 && concepts::weakly_equality_comparable_with<T, decltype(nullable_null_v<T>)>;

137 template <class T>
138 concept nullable = input_nullable<T>
139 && concepts::initializes<std::remove_cvref_t<decltype(nullable_null_v<T>)>, std::remove_cvref_t<T>>
140 && std::is_assignable_v<std::remove_cvref_t<T>&, decltype(nullable_null_v<T>)>;

335 "TFunc return type must be either void or initialize the TNullable type."
336 );

345 template <input_nullable UNullable, class UFunc>
346 [[nodiscard]]
347 constexpr TNullable operator()(UNullable&& closure, UFunc&& func)
348 {
349 if (closure != nullable_null_v<UNullable>)
350 {
351 return std::forward<UNullable>(closure);
352 }
353
354 if constexpr (std::same_as<void, std::invoke_result_t<UFunc>>)
355 {
356 std::invoke(std::forward<UFunc>(func));
357 return nullable_null_v<UNullable>;
358 }
359 else
360 {
361 return std::invoke(std::forward<UFunc>(func));
362 }
363 }

379 template <input_nullable TNullable, class TFunc>
380 struct and_then_func_t<TNullable, TFunc>
381 {
389 template <input_nullable UNullable, class UFunc>
390 [[nodiscard]]
391 constexpr auto operator()(UNullable&& closure, UFunc&& func)
392 {
393 static_assert
394 (
395 std::invocable<UFunc, detail::dereference_type_t<UNullable>>,
396 "TFunc must accept the value returned by TNullables as parameter."
397 );
398
399 using return_t = std::invoke_result_t<UFunc, detail::dereference_type_t<UNullable>>;
400 static_assert(nullable<return_t>, "TFunc must return a nullable type.");
401
402 if (closure != nullable_null_v<UNullable>)
403 {
404 return std::invoke(std::forward<UFunc>(func), value_unchecked(std::forward<UNullable>(closure)));
405 }
406 return return_t{ nullable_null_v<return_t> };
407 }
